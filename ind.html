Let's Search Some Gifs!

Search below to the wonderful world of Gifs!

$( document ).ready(function(){ // Part 1 - Collect User Input Using jQuery Click Listener note we use the class (.) of search_button $('.search_button').on('click', function(){ var userInput = $('#form-value').val().trim(); userInput = userInput.replace(/ /g, "+"); var queryURL = 'http://api.giphy.com/v1/gifs/search?q=' + userInput + '&api_key=dc6zaTOxFJmzC'; $.ajax({url: queryURL, method: 'GET'}).done(function(response){ console.log(response.data); var giphyURL = response.data[0].images.fixed_height.url; console.log(giphyURL) $('#here_is_gif').attr('src', giphyURL); }); $('#reset_button').on('click', function(){ $('#here_is_gif').attr("src",''); }) return false; }) });
